def f_Pois(x, lamb):
    return math.exp(-lamb) * lamb ** x / math.factorial(x)

# 
def F_Pois(x, lamb):
    s = 0
    for i in range(x - 1):
        s += math.exp(-lamb) * lamb ** i / math.factorial(i)
    return s

#   
def rand_Pois(lamb, n):
    nums = []
    for i in range(n):
        p = math.exp(-lamb)
        sp = 0
        sp += p
        u = random.uniform(0,1)
        j = 0
        while u > sp:
            j += 1         # p_j
            p *= lamb / j  # p_j / p_j-1 = lamb / j
            sp += p
        nums.append(j)
    return nums

def F_Rayl(x, sigma):
    return 1 - math.exp(-(x ** 2) / (2 * sigma ** 2))

def inv_f_Rayl(sigma, u):
    return sigma * math.sqrt(-2 * math.log(u))

from bisect import bisect_right

def emp(distr):
    ''' ,  ,   y'''
    n = len(distr)
    yy = []
    count = 0
    for i, val in enumerate(distr):
        if not count:
            prev = val
        if prev == val:
            count += 1
        else:
            yy += count * [i / n]
            prev = val
            count = 1
    yy += count * [i / n]
    def emp_yy(x, distr=distr, yy=yy):
        if x < distr[0]:
            return 0.
        return yy[bisect_right(distr, x) - 1]
    return emp_yy

import math

def p_v(distr, is_discr, par):
    '''
       v=(v1...vN) .  p=(p1...pN) .      
    '''
    p = list()
    v = list()
    n = len(distr)
    if is_discr:
        N = max(distr)
        for i in range(N + 1):
            v.append(distr.count(i))
            p.append(f_Pois(i, par))
    else:
        N = int(math.log(n, 2) + 1)
        end = math.ceil(max(distr))
#         print('. max:', end)
        step0 = math.floor(end / N)
        step = step0 if step0 != 0 else 1
#         print(':', step)
        prev_i = 0
        ind = 0
        for i in range(step, end, step):
            count = 0
            while (ind < n) and (distr[ind] <= i):
                count += 1
                ind += 1
            v.append(count)
            p.append(F_Rayl(i, par) - F_Rayl(prev_i, par))
            prev_i = i
    return p, v
 
def correct_p_v(p, v):
    '''
       p  v:     < 5,    |.
    '''
#     print('-  ', len(v))
    for i in range(len(v)):
        while (i < len(v)) and v[i] < 5:
#             print(v)
            if (i+1) != len(v):
                v[i] += v[i+1]
                p[i] += p[i+1]
                del v[i+1], p[i+1]
            else:
                v[i-1] += v[i]
                p[i-1] += p[i]
                del v[i], p[i]
#     print('  :', len(v))
    return p, v

def chi2(distr, is_discr, par):
    '''   -,  -  ( / )'''
    chi = list()
    n = len(distr)
    p, v = p_v(distr, is_discr, par)
#     p, v = correct_p_v(p, v)
    for i in range(0, len(v)):
        chi.append((v[i] - n * p[i]) ** 2 / (n * p[i]))
    return len(p), sum(chi)

def s_test(comb):
    n1 = len(comb[0])
    n2 = len(comb[1])
    k = math.sqrt(n1 * n2 / (n1 + n2))
    emp1 = emp(comb[0])
    emp2 = emp(comb[1])
    #   ..,  ,   x   
    max12 = max(max(comb[0]), max(comb[1]))
    #    
    my_range = range(max12) if isinstance(max12, int) else np.arange(0, max12, 0.1)
    m_dif = max(list(map(lambda x: abs(emp1(x) - emp2(x)), my_range)))
    return k * m_dif
